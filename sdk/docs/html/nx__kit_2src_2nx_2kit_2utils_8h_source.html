118 template<class MallocFunc>
119 void* mallocAligned(size_t size, size_t alignment, MallocFunc mallocFunc)
120 {
121  if (alignment == 0)
122  return 0;
123  void* ptr = mallocFunc(size + alignment + sizeof(alignment));
124  if (!ptr) //< allocation error
125  return ptr;
126 
127  void* aligned_ptr = (char*) ptr + sizeof(alignment); //< Leaving place to save misalignment.
128  const size_t misalignment = alignment - (((uintptr_t) aligned_ptr) % alignment);
129  memcpy((char*) ptr + misalignment, &misalignment, sizeof(misalignment)); //< Save misalignment.
130  return (char*) aligned_ptr + misalignment;
131 }
